```verilog
module comb_sensitivity (
  input  wire a,
  input  wire b,
  input  wire c,
  input  wire sel,
  output reg out
);

always @(sel or a) begin
  case (sel)
    1'b0: begin
      out = a & b;
    end
    1'b1: begin
      out = b | c;
    end
    default: out = 1'b0; // safe default value for blocking assignment
  endcase
end

endmodule
```
