```verilog
module alu_ops (
  input [7:0] a,
  input [7:0] b,
  input [2:0] opcode,
  output reg [7:0] result,
  output reg carry_out,
  output reg zero
);

  always @(*) begin
    case (opcode)
      3'b000: begin
        result = a + b;
        carry_out = 0;
        zero = (result == 0);
      end
      3'b001: begin
        result = a - b;
        carry_out = (a < b);
        zero = (result == 0);
      end
      3'b010: begin
        result = a & b;
        carry_out = 0;
        zero = (result == 0);
      end
      3'b011: begin
        result = a | b;
        carry_out = 0;
        zero = (result == 0);
      end
      3'b100: begin
        result = a ^ b;
        carry_out = 0;
        zero = (result == 0);
      end
      default: begin
        result = 0;
        carry_out = 0;
        zero = 0;
      end
    endcase
  end

endmodule
```
