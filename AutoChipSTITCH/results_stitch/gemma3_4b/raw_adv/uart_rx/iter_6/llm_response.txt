module uart_rx (
  input clk,
  input rst,
  input rx,
  output reg [7:0] rx_data,
  output reg data_valid
);

  parameter CLKS_PER_BIT = 4;

  // FSM States
  typedef enum logic [1:0] {
    IDLE,
    RECEIVING
  } state_type;
  reg state;

  // Internal Signals
  reg [7:0] bit_buffer;
  reg bit_count;
  reg byte_complete;

  // Combinational Logic
  always @(posedge clk or negedge rst) begin
    if (rst) begin
      state <= IDLE;
      bit_count <= 0;
      byte_complete <= 0;
      rx_data <= 0;
      data_valid <= 0;
    end else begin
      case (state)
        IDLE: begin
          // Idle state: Wait for start bit (low)
          if (rx == 0) begin
            state <= RECEIVING;
            bit_count <= 0;
          end else begin
            // Stay in idle state if no start bit
            state <= IDLE;
          end
        end
        RECEIVING: begin
          // Receiving state: Sample bit and accumulate
          if (bit_count < CLKS_PER_BIT) begin
            bit_buffer[7 - bit_count] <= rx;
            bit_count <= bit_count + 1;
          end else begin
            // Byte complete
            state <= IDLE;
            byte_complete <= 1;
            data_valid <= 1;
          end
        end
        default: begin
          state <= IDLE;
        end
      endcase
    end
  end

endmodule
