module lfsr_8bit(input clk, rst, load, enable, input [7:0] seed, output reg [7:0] lfsr_out);

      lfsr_out <= 8'hFF;
    else if (load)
      lfsr_out <= seed;
    else if (enable)
      lfsr_out <= {lfsr_out[6:0], lfsr_out[7] ^ lfsr_out[0]};  // XOR feedback with lfsr_out[7]
  end
endmodule

When reset (rst) is high, the lfsr_out is set to 8'hFF. When load (load) is high, the lfsr_out is loaded with the seed value. When enable (enable) is high, the LFSR performs a right-shift operation and XOR feedback.
