```verilog
module spi_master_8bit(input clk, rst, start, input [7:0] mosi_data, input miso, 
                       output reg sclk, cs_n, mosi, done, 
                       output reg [7:0] miso_capture);
    parameter CLK_DIV = 4;
    
    // States definition
    typedef enum {IDLE=2'd0, ACTIVE=2'd1, DONE_ST=2'd2} state_t;
    state_t state, next_state;
    
    // Internal registers
    reg [7:0] shift_out, shift_in; 
    reg [3:0] clk_cnt, bit_cnt;
    
    always @(posedge clk or posedge rst) begin
        if (rst) begin
            state <= IDLE;
            cs_n <= 1'b1;
            sclk <= 1'b0;
            mosi <= 1'b0;
            done <= 1'b0;
            miso_capture <= 8'd0;
        end 
        else begin
            state <= next_state;
            
            case (state)
                IDLE: begin
                    cs_n <= 1'b1;
                    sclk <= 1'b0;
                    done <= 1'b0;
                end
                
                ACTIVE: begin
                    if(clk_cnt < CLK_DIV) begin // First half
                        sclk <= 1'b0;
                        mosi <= shift_out[7];
                    end 
                    else if (clk_cnt == CLK_DIV - 1) begin // At clk_cnt==CLK_DIV-1
                        sclk <= 1'b1;
                    end 
                    else begin // Second half
                        sclk <= 1'b1;
                        
                        if(clk_cnt == (2*CLK_DIV - 2)) begin
                            shift_in <= {shift_in[6:0], miso};
                        end 
                        else if(clk_cnt == (2*CLK_DIV - 1)) begin
                            sclk <= 1'b0;
                            shift_out <= {shift_out[6:0], 1'b0};
                            clk_cnt <= 4'd0;
                            bit_cnt <= bit_cnt + 1'b1;
                        end 
                    end 
                end
                
                DONE_ST: begin
                    done <= 1'b1; // Assert done for one clock cycle
                    miso_capture <= shift_in;
                    next_state <= IDLE;
                end
            endcase
            
            if(start) begin
                cs_n <= 1'b0;
                shift_out <= mosi_data;
                bit_cnt <= 4'd0;
                clk_cnt <= 4'd0;
                next_state <= ACTIVE;
            end 
        end
    end
endmodule
```
This module implements the SPI master mode 0 (CPOL=0, CPHA=0), 8-bit MSB-first. The state machine is implemented using combinational logic and sequential logic to handle the clock division and data shifting operations. It also handles the reset condition by setting all outputs to their initial states when a reset signal is asserted.
