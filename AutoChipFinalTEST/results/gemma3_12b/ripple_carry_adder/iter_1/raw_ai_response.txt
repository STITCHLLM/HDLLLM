```verilog
module ripple_carry_adder (
  input [3:0] A,
  input [3:0] B,
  input cin,
  output [3:0] Sum,
  output cout
);

  wire [3:0] sum_wires;
  wire [3:0] carry_wires;

  full_adder fa0 (
    .a(A[0]),
    .b(B[0]),
    .cin(cin),
    .sum(sum_wires[0]),
    .cout(carry_wires[0])
  );

  full_adder fa1 (
    .a(A[1]),
    .b(B[1]),
    .cin(carry_wires[0]),
    .sum(sum_wires[1]),
    .cout(carry_wires[1])
  );

  full_adder fa2 (
    .a(A[2]),
    .b(B[2]),
    .cin(carry_wires[1]),
    .sum(sum_wires[2]),
    .cout(carry_wires[2])
  );

  full_adder fa3 (
    .a(A[3]),
    .b(B[3]),
    .cin(carry_wires[2]),
    .sum(sum_wires[3]),
    .cout(cout)
  );

  assign Sum = sum_wires;

endmodule
```
