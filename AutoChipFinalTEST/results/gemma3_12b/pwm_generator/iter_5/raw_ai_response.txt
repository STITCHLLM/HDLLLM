```verilog
module pwm_generator (
  input clk,
  input rst,
  input [7:0] duty_cycle,
  output reg pwm_out
);

  reg [7:0] counter;

  always @(posedge clk) begin
    if (rst)
      counter <= 8'd0;
    else
      counter <= counter + 1'b1;
  end

  always @(posedge clk) begin
    if (rst)
      pwm_out <= 1'b0;
    else if (counter < duty_cycle)
      pwm_out <= 1'b1;
    else
      pwm_out <= 1'b0;
  end

endmodule
```
