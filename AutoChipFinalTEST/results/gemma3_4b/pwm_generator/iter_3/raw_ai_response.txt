```verilog
module pwm_generator (
  input clk,
  input rst,
  input [7:0] duty_cycle,
  output wire pwm_out
);

  reg [7:0] counter;
  wire pwm_out;

  always @(posedge clk) begin
    if (rst) begin
      counter <= 8'b0;
    end else begin
      counter <= counter + 1'b1;
    end
  end

  assign pwm_out = (counter < duty_cycle);

endmodule
```
