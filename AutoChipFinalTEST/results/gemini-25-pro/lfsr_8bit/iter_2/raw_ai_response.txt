```verilog
module lfsr_8bit (
    input clk,
    input rst,
    input load,
    input enable,
    input [7:0] seed,
    output reg [7:0] lfsr_out
);

    always @(posedge clk) begin
        if (rst)
            lfsr_out <= 8'hFF;
        else if (load)
            lfsr_out <= seed;
        else if (enable)
            // Corrected feedback logic for polynomial x^8 + x^6 + x^5 + x^4 + 1
            // Taps are at positions 6, 5, 4.
            lfsr_out <= {
                lfsr_out[0],              // new bit[7] = feedback
                lfsr_out[7],              // new bit[6]
                lfsr_out[6]^lfsr_out[0],  // new bit[5] (tap x^6)
                lfsr_out[5]^lfsr_out[0],  // new bit[4] (tap x^5)
                lfsr_out[4]^lfsr_out[0],  // new bit[3] (tap x^4)
                lfsr_out[3:1]             // new bit[2:0]
            };
    end

endmodule
```
