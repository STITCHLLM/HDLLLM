```verilog
module uart_tx #(
    parameter BAUD_DIV = 104
) (
    input clk,
    input rst,
    input start,
    input [7:0] data_in,

    output reg tx_out,
    output reg busy,
    output reg done
);

// FSM States
localparam [2:0]
    IDLE  = 3'd0,
    START = 3'd1,
    DATA  = 3'd2,
    STOP  = 3'd3,
    DONE  = 3'd4;

// Internal Registers
reg [2:0] state;
reg [7:0] shift_reg;
reg [7:0] baud_cnt; // Declared as [7:0] as per requirement
reg [3:0] bit_cnt;

always @(posedge clk or posedge rst) begin
    if (rst) begin
        state <= IDLE;
        tx_out <= 1'b1;
        busy <= 1'b0;
        done <= 1'b0;
        shift_reg <= 8'h00;
        baud_cnt <= BAUD_DIV; // Initialize baud_cnt for countdown
        bit_cnt <= 4'h0;
    end else begin
        // 'done' is only high for one cycle in the DONE state,
        // so default it to low and set it high only when needed.
        done <= 1'b0;

        case (state)
            IDLE: begin
                tx_out <= 1'b1; // Default idle state is high
                busy <= 1'b0;   // Not busy in idle
                baud_cnt <= BAUD_DIV; // Reset baud_cnt for next transmission
                bit_cnt <= 4'h0;   // Reset bit_cnt
                if (start) begin
                    state <= START;
                    shift_reg <= data_in; // Latch data_in
                    tx_out <= 1'b0; // Assert start bit immediately on transition
                    busy <= 1'b1;   // Assert busy immediately on transition
                end
            end

            START: begin
                // tx_out is 0, busy is 1 (set on transition from IDLE)
                if (baud_cnt == 1) begin // Last cycle of BAUD_DIV period
                    state <= DATA;
                    baud_cnt <= BAUD_DIV; // Reset baud_cnt for next bit
                    bit_cnt <= 4'h0;   // Initialize bit_cnt for data bits (0 to 7)
                end else begin
                    baud_cnt <= baud_cnt - 1;
                end
            end

            DATA: begin
                tx_out <= shift_reg[0]; // Transmit LSB first
                busy <= 1'b1; // Keep busy asserted
                if (baud_cnt == 1) begin // Last cycle of BAUD_DIV period
                    baud_cnt <= BAUD_DIV; // Reset baud_cnt for next bit
                    shift_reg <= shift_reg >> 1; // Shift data for next bit
                    if (bit_cnt == 7) begin // After 8 bits (0 to 7)
                        state <= STOP;
                    end else begin
                        bit_cnt <= bit_cnt + 1;
                    end
                end else begin
                    baud_cnt <= baud_cnt - 1;
                end
            end

            STOP: begin
                tx_out <= 1'b1; // Stop bit is 1
                busy <= 1'b1; // Keep busy asserted
                if (baud_cnt == 1) begin // Last cycle of BAUD_DIV period
                    state <= DONE;
                    baud_cnt <= BAUD_DIV; // Reset baud_cnt
                end else begin
                    baud_cnt <= baud_cnt - 1;
                end
            end

            DONE: begin
                tx_out <= 1'b1; // Keep high
                busy <= 1'b0;
                done <= 1'b1; // Assert done for one cycle
                state <= IDLE; // Go back to IDLE
            end

            default: begin // Should not happen, but good for completeness
                state <= IDLE;
                tx_out <= 1'b1;
                busy <= 1'b0;
                done <= 1'b0;
                shift_reg <= 8'h00;
                baud_cnt <= BAUD_DIV;
                bit_cnt <= 4'h0;
            end
        endcase
    end
end

endmodule
```
