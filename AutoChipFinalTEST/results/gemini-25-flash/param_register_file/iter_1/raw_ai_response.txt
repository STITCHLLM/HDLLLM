module param_register_file #(
  parameter N = 4,
  parameter W = 8
) (
  input             clk,
  input             wr_en,
  input  [1:0]      wr_addr,
  input  [1:0]      rd_addr,
  input  [W-1:0]    wr_data,
  output wire [W-1:0] rd_data
);

  // Internal memory declaration
  reg [W-1:0] mem [0:N-1];

  // Synchronous write operation
  always @(posedge clk) begin
    if (wr_en) begin
      mem[wr_addr] <= wr_data;
    end
  end

  // Combinational read operation
  assign rd_data = mem[rd_addr];

endmodule
