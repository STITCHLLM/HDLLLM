```verilog
module uart_tx (
    input clk,
    input rst,
    input start,
    input [7:0] data_in,
    output reg tx_out,
    output reg busy,
    output reg done
);

parameter BAUD_DIV = 104;

// FSM states
reg [2:0] state;
`define IDLE  3'd0
`define START 3'd1
`define DATA  3'd2
`define STOP  3'd3
`define DONE  3'd4

// Internal registers
reg [7:0] shift_reg;  // data shift register
reg [7:0] baud_cnt;   // baud counter
reg [3:0] bit_cnt;    // bit index 0..7

always @(posedge clk or posedge rst) begin
    if (rst) begin
        state       <= `IDLE;
        tx_out      <= 1'b1;
        busy        <= 1'b0;
        done        <= 1'b0;
        shift_reg   <= 8'h00;
        baud_cnt    <= 8'h00;
        bit_cnt     <= 4'h00;
    end else begin
        case (state)
            `IDLE: begin
                tx_out <= 1'b1;
                busy   <= 1'b0;
                done   <= 1'b0;
                if (start) begin
                    shift_reg <= data_in;
                    baud_cnt  <= 8'h00;
                    bit_cnt   <= 4'h00;
                    state     <= `START;
                end else begin
                    state     <= `IDLE; // Stay in IDLE
                end
            end

            `START: begin
                tx_out <= 1'b0; // Start bit is 0
                busy   <= 1'b1;
                done   <= 1'b0;
                if (baud_cnt == BAUD_DIV - 1) begin
                    baud_cnt <= 8'h00;
                    state    <= `DATA;
                end else begin
                    baud_cnt <= baud_cnt + 1'b1;
                    state    <= `START; // Stay in START
                end
            end

            `DATA: begin
                busy   <= 1'b1;
                tx_out <= shift_reg[0]; // Transmit LSB first
                done   <= 1'b0;
                if (baud_cnt == BAUD_DIV - 1) begin
                    baud_cnt  <= 8'h00;
                    shift_reg <= {1'b0, shift_reg[7:1]}; // Shift right, MSB gets 0
                    if (bit_cnt == 4'd7) begin // After 8 data bits (0 to 7)
                        bit_cnt <= 4'h00; // Reset for next transmission
                        state   <= `STOP;
                    end else begin
                        bit_cnt <= bit_cnt + 1'b1;
                        state   <= `DATA; // Stay in DATA
                    end
                end else begin
                    baud_cnt <= baud_cnt + 1'b1;
                    state    <= `DATA; // Stay in DATA
                end
            end

            `STOP: begin
                tx_out <= 1'b1; // Stop bit is 1
                busy   <= 1'b1;
                done   <= 1'b0;
                if (baud_cnt == BAUD_DIV - 1) begin
                    baud_cnt <= 8'h00;
                    state    <= `DONE;
                end else begin
                    baud_cnt <= baud_cnt + 1'b1;
                    state    <= `STOP; // Stay in STOP
                end
            end

            `DONE: begin
                done   <= 1'b1;
                busy   <= 1'b0;
                tx_out <= 1'b1; // Keep high for idle
                state  <= `IDLE;
            end

            default: begin // Should not happen, but good practice for FSMs
                state       <= `IDLE;
                tx_out      <= 1'b1;
                busy        <= 1'b0;
                done        <= 1'b0;
                baud_cnt    <= 8'h00;
                bit_cnt     <= 4'h00;
            end
        endcase
    end
end

endmodule
```
