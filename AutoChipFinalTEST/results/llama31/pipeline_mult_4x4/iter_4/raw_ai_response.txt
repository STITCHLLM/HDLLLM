```verilog
module pipeline_mult_4x4 (
  input clk,
  input [3:0] a, b,
  output reg [7:0] product
);

reg [3:0] a_r, b_r;

always @(posedge clk) begin
  #1 a_r     <= a;           // Stage 1: latch inputs with delay
  #1 b_r     <= b;
  #2 product <= a_r * b_r;   // Stage 2: multiply the latched values with delay
end

endmodule
```
