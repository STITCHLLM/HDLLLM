module ripple_carry_adder (
    input [3:0] A,
    input [3:0] B,
    input       cin,
    output [3:0] Sum,
    output      cout
);

    // Outputs driven by sub-module instantiations MUST be declared as wire.
    wire [3:0] Sum;
    wire       cout;

    // Intermediate carry wires
    wire c1, c2, c3;

    // Instantiate full_adder four times
    // fa0: A[0], B[0], cin -> Sum[0], c1
    full_adder fa0 (
        .a   (A[0]),
        .b   (B[0]),
        .cin (cin),
        .sum (Sum[0]),
        .cout(c1)
    );

    // fa1: A[1], B[1], c1 -> Sum[1], c2
    full_adder fa1 (
        .a   (A[1]),
        .b   (B[1]),
        .cin (c1),
        .sum (Sum[1]),
        .cout(c2)
    );

    // fa2: A[2], B[2], c2 -> Sum[2], c3
    full_adder fa2 (
        .a   (A[2]),
        .b   (B[2]),
        .cin (c2),
        .sum (Sum[2]),
        .cout(c3)
    );

    // fa3: A[3], B[3], c3 -> Sum[3], cout
    full_adder fa3 (
        .a   (A[3]),
        .b   (B[3]),
        .cin (c3),
        .sum (Sum[3]),
        .cout(cout)
    );

endmodule
