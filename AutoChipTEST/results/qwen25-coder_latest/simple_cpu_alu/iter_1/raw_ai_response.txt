```verilog
module simple_cpu_alu (
    input [3:0] A,
    input [3:0] B,
    input [1:0] op,
    output [3:0] result,
    output zero
);

reg [3:0] result_reg;
wire zero_wire;

always @(*) begin
    case (op)
        2'b00: result_reg = A + B; // ADD
        2'b01: result_reg = A - B; // SUB
        2'b10: result_reg = A & B; // AND
        2'b11: result_reg = A | B; // OR
        default: result_reg = 4'b0000;
    endcase
end

assign zero_wire = (result_reg == 4'b0000);

assign result = result_reg;
assign zero = zero_wire;

endmodule
```
