```verilog
module simple_cpu_alu (
    input [3:0] A,
    input [3:0] B,
    input [1:0] op,
    output [3:0] result,
    output zero
);

reg [3:0] result;

wire zero = (result == 4'b0000);

always @(op or A or B)
begin
    case (op)
        2'b00: result = A + B;
        2'b01: result = A - B;
        2'b10: result = A & B;
        2'b11: result = A | B;
        default: result = 4'b0000; // Default to zero for unknown opcodes
    endcase
end

endmodule
```
