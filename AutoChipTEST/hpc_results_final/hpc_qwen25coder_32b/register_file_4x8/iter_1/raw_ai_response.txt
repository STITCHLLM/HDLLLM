```verilog
module register_file_4x8 (
    input clk,
    input we,
    input [1:0] waddr,
    input [1:0] raddr1,
    input [1:0] raddr2,
    input [7:0] wdata,
    output [7:0] rdata1,
    output [7:0] rdata2
);

    reg [7:0] reg_file [3:0];

    always @(posedge clk) begin
        if (we) begin
            reg_file[waddr] <= wdata;
        end
    end

    assign rdata1 = reg_file[raddr1];
    assign rdata2 = reg_file[raddr2];

endmodule
```
