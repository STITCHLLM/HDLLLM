 ```verilog
module alu_8bit (
    input [7:0] A,
    input [7:0] B,
    input [2:0] op,
    output [7:0] result,
    output zero
);

reg [7:0] result;
wire zero;

always @(*) begin
    case (op)
        3'b000: result = A + B; // ADD
        3'b001: result = A - B; // SUB
        3'b010: result = A & B; // AND
        3'b011: result = A | B; // OR
        3'b100: result = A ^ B; // XOR
        3'b101: result = ~A;    // NOT_A
        3'b110: result = A << 1; // SHL
        3'b111: result = A >> 1; // SHR
        default: result = 8'b0;  // Default case
    endcase
end

assign zero = (result == 8'b0);

endmodule
```
